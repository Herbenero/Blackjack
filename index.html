// Modify the scoreRound function to include House earning 3 points if House wins the round function scoreRound() { const house = state.players.find(p => p.isHouse); const nonHouse = state.players.filter(p => !p.isHouse); const roundSummary = []; roundSummary.push(`Round ${state.round}: House was ${escapeHtml(house.name)} with ${house.busted ? 'BUST' : house.currentRoundScore}.`); if (house.busted) { const survivors = nonHouse.filter(p => !p.busted); const tiedBust = nonHouse.filter(p => p.busted); survivors.forEach(p => { p.overall += 3; p.currentResult = 'Beat House'; }); tiedBust.forEach(p => { p.currentResult = 'Bust'; }); house.currentResult = 'House Bust'; if (survivors.length) { roundSummary.push(`${survivors.map(p => escapeHtml(p.name)).join(', ')} earned 3 pts because the House busted.`); } if (tiedBust.length) { roundSummary.push(`${tiedBust.map(p => escapeHtml(p.name)).join(', ')} also busted and earned 0 pts.`); } } else { // House not busted let houseWon = true; // Assume house wins unless a player beats or ties nonHouse.forEach(player => { if (player.busted) { player.currentResult = 'Bust'; return; } if (player.currentRoundScore > house.currentRoundScore) { player.overall += 3; player.currentResult = 'Beat House'; houseWon = false; } else if (player.currentRoundScore === house.currentRoundScore) { player.overall += 1; player.currentResult = 'Tied House'; houseWon = false; } else { player.currentResult = 'Lost to House'; } }); if (houseWon) { house.overall += 3; // House earns 3 points for winning the round house.currentResult = 'House Won'; roundSummary.push(`House won the round and earned 3 pts.`); } else { // House did not win outright, check if any ties const tiedPlayers = nonHouse.filter(p => !p.busted && p.currentRoundScore === house.currentRoundScore); if (tiedPlayers.length) { house.overall += 1; house.currentResult = 'House Tie'; roundSummary.push(`${tiedPlayers.map(p => escapeHtml(p.name)).join(', ')} tied the House and each earned 1 pt; the House also earned 1 pt.`); } else { house.currentResult = 'House Lost'; } } const winners = nonHouse.filter(p => p.currentResult === 'Beat House'); const losers = nonHouse.filter(p => p.currentResult === 'Lost to House'); const busts = nonHouse.filter(p => p.currentResult === 'Bust'); if (winners.length) { roundSummary.push(`${winners.map(p => escapeHtml(p.name)).join(', ')} earned 3 pts.`); } if (losers.length) { roundSummary.push(`${losers.map(p => escapeHtml(p.name)).join(', ')} scored 0 pts toward overall.`); } if (busts.length) { roundSummary.push(`${busts.map(p => escapeHtml(p.name)).join(', ')} busted and earned 0 pts.`); } } state.logs.unshift(roundSummary.join(' ')); }
